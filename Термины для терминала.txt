pwd #посмотреть где я
mkdir #создать папку
cd #зайти в папку
touch #создать файл без текста
cat > #создать файл с добавленим текста
cat #просмотреть содержание файла
ctrl+D #завершить написание текста в команде cat >
ls #вывод списка содержимого папки
ls -la #выводит список содержимого, показывая скрытые файлы, их владельцев и т.д.
cp #копировать файл
mv #переместить файл
less #прочитать/просмотреть файл(без редактирования), быстрее чем vim т.к. не просматривает файл перед его открытием
less + #показать первые несколько строк(less +2) показывает первые 2 строки
less G #просмотр последней строки текста
grep #поиск строк в нескольких входных данных строки
vim # текстовый редактор
:qw #сохранить и выйти из vim
:q #выйти из vim/less
i #начать редактирование в vim
date #просмотр даты и времени
